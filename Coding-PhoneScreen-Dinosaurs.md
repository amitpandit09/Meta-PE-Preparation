import math
import heapq
def CSVDinosaurs(path1, path2):
    g = 9.8
    stride_map = {}
    with open(path2, 'r') as f:
        line = f.readline()
        line = f.readline()
        while line:
            tmp = line.split(",")
            name, stride, stance = tmp
            if stance.strip() == "bipedal":
                stride_map[name] = float(stride)
            line = f.readline()

    outmap = {}
    with open(path1, 'r') as f:
        line = f.readline()
        line = f.readline()
        while line:
            tmp = line.split(",")
            name, leg, diet = tmp
            leg = float(leg)
            if name in stride_map:
                stride = stride_map[name]
                outmap[name] = (stride/leg-1)*math.sqrt(leg*g)
                print(outmap[name])
            line = f.readline()

    res = []
    for name in outmap:
        res.append((-outmap[name], name))
    heapq.heapify(res)
    while res:
        speed, name = heapq.heappop(res)
        print(name + ", " + str(-speed))

path1 = "dataset1.csv"
path2 = "dataset2.csv"
CSVDinosaurs(path1, path2)
